
```shell

docker run -d --name db2 -p 50000:50000 -e DB2INST1_PASSWORD=000000 -e LICENSE=accept ibmoms/db2express-c:latest db2start

a. 进入镜像 
docker exec -it db2 /bin/bash

b.在镜像里面 添加用户 
useradd -g users -d /home/mtdb2dev -s /bin/bash -m mtdb2dev

c.修改用户密码
passwd mtdb2dev

d.切换到 db2inst1用户
su db2inst1

e.链接目标数据库 
db2 connect to AUDIT

f.给数据库授权用户 
db2 grant dbadm on database to user mtdb2dev



# 进入db2交互式命令行
db2

```
